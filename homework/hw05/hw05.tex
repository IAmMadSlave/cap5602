\begin{homeworkProblem}
    \textbf{8.6} Which of the following are valid (necessarily true) sentences?
    \begin{enumerate}[label=\alph*.]
        \item $(\exists x\ x=x)\Rightarrow (\forall y\ \exists z\ y=z)$
        \item $\forall x\ P(x)\lor \neg P(x)$
        \item $\forall x\ Smart(x) \lor (x=x)$
    \end{enumerate}
    \vspace{5 mm}
    \begin{enumerate}[label=\alph*.]
        \item Valid. 
        \item Valid. Either $P(x)$ is true or it is not.
        \item Valid. The right hand side has $x=x$ which will always be true,
            thus the sentence will always be true.
    \end{enumerate}
\end{homeworkProblem}

\begin{homeworkProblem}
    \textbf{8.10} Consider a vocabulary with the following symbols:
    \begin{itemize}
        \item $Occupation(p,o):$ Predicate. Person $p$ has occupation $o$.
        \item $Customer(p1,p2:$ Predicate. Person $p1$ is a customer of person
            $p2$.
        \item $Boss(p1,p2):$ Predicate. Person $p1$ is a boss of person $p2$.
        \item $Doctor, Surgeon, Lawyer, Actor:$ Constants denoting occupations.
        \item $Emily, Joe:$ Constants denoting people.
    \end{itemize}

    Use these symbols to write the following assertion in first-order logic:
    \begin{enumerate}[label=\alph*.]
        \item Emily is either a surgeon or a lawyer.
        \item Joe is an actor, but he also holds another job.
        \item All surgeons are doctors.
        \item Joe does not have a lawyer (i.e., is not a customer of any
            lawyer).
        \item Emily has a boss who is a lawyer.
        \item There exists a lawyer all of whose customers are doctors.
        \item Every surgeon has a lawyer.
    \end{enumerate}
    \vspace{3 mm}
    \begin{enumerate}[label=\alph*.]
        \item $Occupation(Emily,Surgeon) \lor Occupation(Emily,Lawyer)$
        \item $Occupation(John,Actor) \land \exists o\ \ne Actor \land
            Occupation(John,o)$
        \item $\forall p\ Occupation(p,Surgeon) \Rightarrow
            Occupation(p,Doctor)$
        \item $\neg \exists p\ Customer(John,p) \land Occupation(p,Lawyer)$
        \item $\exists p\ Boss(p,Emily) \land Occupation(p,Lawyer)$
        \item $\exists p\ Occupation(p,Lawyer) \land \forall q\ Customer(q,p)
            \Rightarrow Occupation(q,Doctor)$
        \item $\forall p\ Occupation(p,Surgeon) \Rightarrow \exists q\
            Occupation(q,Lawyer) \land Customer(p,q)$
    \end{enumerate}

\end{homeworkProblem}
\clearpage

\begin{homeworkProblem}

        \item No two people have the same social security number.\\
            $\neg \exists x,y,n\ Person(x) \land Person(y) \Rightarrow
            [HasSS\#(x,n) \land HasSS\#(y,n)]$
        \item John's social security number is the same as Mary's.\\
            $\exists n\ HasSS\#(John,n) \land HasSS\#(Mary,x)$
        \item Everyone's social security number has nine digits.\\
            $\forall x,n\ Person(x) \Rightarrow [HasSS\#(x,n) \land
            Digits(n,9)]$
        \item Rewrite each of the above (uncorrected) sentence using a function
            symbol $SS\#$ instead of the predicate $HasSS\#$.
    \end{enumerate}
    \vspace{3 mm}
    \begin{enumerate}[label=\alph*.]
        \item Incorrect.\\
            $\neg \exists x,y,n\ Person(x) \land Person(y) \Rightarrow
            [HasSS\#(x,n) \land HasSS\#(y,n)]$
        \item Correct.
        \item Incorrect.\\
            $\forall x,n\ Person(x) \land HasSS\#(x,n) \Rightarrow Digits(n,9)$
        \item $\neg \exists x,y\ Person(x) \land Person(y) \Rightarrow [SS\#(x)
            = SS\#(y)]$\\
            $SS\#(John) = SS\#(Mary)$\\
            $\forall x\ Person(x) \Rightarrow Digits(SS\#(x),9)$
    \end{enumerate}
\end{homeworkProblem}

\begin{homeworkProblem}
    \textbf{9.6} Write down logical representations for the following
    sentences, suitable for use with Generalized Modus Ponens:
    \begin{enumerate}[label=\alph*.]
        \item Horses, cows, and pigs are mammals.
        \item An offspring of a horse is a horse.
        \item Bluebeard is a horse.
        \item Bluebeard is Charlie's parent.
        \item Offspring and parent are inverse relations.
        \item Every mammal has a parent.
    \end{enumerate}
    \vspace{5 mm}
    \begin{enumerate}[label=\alph*.]
        \item $Horse(x) \Rightarrow Mammal(x)$\\
              $Cow(x) \Rightarrow Mammal(s)$\\
              $Pig(x) \Rightarrow Mammal(s)$
        \item $Offspring(a,b) \land Horse(b) \Rightarrow Horse(a)$
        \item $Horse(Bluebeard)$
        \item $Parent(Bluebeard,Charlie)$
        \item $Offspring(a,b) \Leftrightarrow Parent(b,a)$
    \end{enumerate}

\end{homeworkProblem}
